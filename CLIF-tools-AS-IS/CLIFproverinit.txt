:- initialization(once(((run ; true), halt))).

% Parameters
project("BCHANGE").			% Works only with 'X-cf.txt' and 'X-nsat.txt' in format generated by BFO2020style-CLIFparser.cl 
							% 	with 'X' replaced by the name provided as text in the project("...") parameter.
toprove("TEST").			% Name of the ontology to test consistency of with the base ontologies.
inputdata("*-input.txt").	% should be in the 't/c-format' of the 'database' section as used in BMI714 'RTreasoning"
max_threshold(7,2,1).		% maximum threshold levels to terminate prover
							% 	arg1=allowed clause length for clause selection (except for input clauses)
							% 	arg2=allowed level of functions in functions 	(e.g. [sk1, ..., [sk4, ...]]  )
							% 	arg3=allowed auto-recursion level of function	(e.g. [sk2, ..., [sk2, ...]]  )
w_threshold(1,4,9).			% weight of max_threshold values (mta_n) combined in one single value as follows:
							%	w = (arg1*mta_1) + (arg2*mta_2) + (arg3*mta_3)
bws([a,p]).					% list type of clauses allowed to be backward simplified
							% 	arg1=[x, y, ...], with allowed symbols
							%	a: active clauses
							%	p: passive clauses
							%	u: unprocessed clauses
strategy(sos).				% specify reasoning strategy to use. Possibilities:
							%	sos:	set of support
db_ratio(7,5).				% sets the ratio of depth-first, breadth-first.
							%	arg1=number of cycles depth- and breadth-first together
							%	arg2=number of depth-first cycles
							%   for example: db_ratio(7,5) indicates 5 depth first cycles followed by 2 breadth-first 
opr([vrp,cdp,cdo,coo,fdp,fsp,pdo,fdo,foo,fso,fuf,ide,pnd,pno]).	% ascending list of listnames for term ordering.
							% these listnames are reserved names to be used as such and for nothing else
							% none my be deleted, or added, but the order may be changed to change the term ordering
							%	vrp:	variable proxies
							%	cdp:	constants only used in data and/or toprove input
							%	cdo:	constants used in both base theory and data input
							%	coo:	constants only used in base theory
							%	fdp:	non-skolem functions only used in data and/or toprove input
							%	fsp:	skolem functions only used in toprove input
							%	pdo:	base theory predicate names used in data and toprove input
							%	fdo:	functions used in both base theory and data input 
							%	foo:	non-skolem functions only used in base theory 
							%	fso:	skolem functions only used in base theory 
							%	fuf:	function symbol of the only unary input function if any at all
							%	ide:	'identical'
							%	pnd:	predicate names not in pdo
							%	pno:	predicate names not in base theory
